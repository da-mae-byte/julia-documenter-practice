文字列操作

str_hoge = 'python'
str_date = '20210123'

抽出

インデックスを指定（スライス）

print(str_hoge[:3])  # pyt
print(str_hoge[1:5]) # ytho
print(str_hoge[3:])  # hon

yr = str_date[0:4]  # 2021
mo = str_date[4:6]  # 01
dy = str_date[6:8]  # 23
(以下のように考える)
   +-+-+-+-+-+-+
   |p|y|t|h|o|n|
   +-+-+-+-+-+-+
   0 1 2 3 4 5 6

分割

str_datetime = '2021/01/23 12:34:56'

date = str_datetime.split(' ')[0]
time = str_datetime.split(' ')[1]

str_yr = date.split('/')[0]
str_mo = date.split('/')[1]
str_dy = date.split('/')[2]

str_hr = date.split(':')[0]
str_mi = date.split(':')[1]
str_sc = date.split(':')[2]
str_conf  = 'abc   0  1  2\n'
str_conf += 'def   3 14  6\n'
str_conf += 'xyz  20 51 32'

# 悪い例
print(str_conf.split(' '))

# 良い例
print(str_conf.split())

# リスト内包表記で配列作成
list_conf = [s.split() for s in str_conf.split('\n')]
print(list_conf)

# ['abc', '', '', '0', '', '1', '', '2\ndef', '', '', '3', '14', '', '6\nxyz', '', '20', '51', '32']
# ['abc', '0', '1', '2', 'def', '3', '14', '6', 'xyz', '20', '51', '32']
# [['abc', '0', '1', '2'], ['def', '3', '14', '6'], ['xyz', '20', '51', '32']]

部分削除

両側／片側だけ削除

str_hoge = 'ABCABCA'
print(str_hoge.strip('A'))   # BCABC
print(str_hoge.rstrip('A'))  # ABCABC
print(str_hoge.lstrip('A'))  # BCABCA

端っこから該当する文字を順番に消しているだけなので、 順番を変えても結果は同じ

str_hoge = 'ABCABCA'
print(str_hoge.lstrip('AB'))  # CABCA
print(str_hoge.lstrip('BA'))  # CABCA

このせいでこんな想定外の結果が出ることもある

path = 'home/work/python/test.py'
print(path.lstrip('home/work/python'))  # st.py (tとeがデイレクトリのパスに含まれているため)

こういうときはreplaceを使う

print(path.replace('home/work/python', '')  # /test.py

フォーマット

x = 24

配列操作

<br>

作成と要素追加

list_hoge = [0, 1, 2, 3, 4]

print(list_hoge)
[0, 1, 2, 3, 4]

print(list_hoge.append(5))
[0, 1, 2, 3, 4, 5]

print(list_hoge.append([5, 6, 7, 8]))
[0, 1, 2, 3, 4, [5, 6, 7, 8]]

print(list_hoge.extend([5, 6, 7, 8]))
[0, 1, 2, 3, 4, 5, 6, 7, 8]

print(list_hoge.insert(0, 100))
[100, 0, 1, 2, 3, 4]

print(list_hoge.insert(-1, 200))
[0, 1, 2, 3, 200, 4]

削除

すべての要素をクリア

list_hoge.clear()

指定した位置の要素を削除し、値を取得

list_hoge.pop(0)
list_hoge.pop(-1)

値を検索して要素を削除（最初の1つのみ）

list_fuga = [0, 1, 2, 3, 1, 4, 5]
list_fuga.remove(1)
print(list_fuga)  # [0, 2, 3, 1, 4, 5]

スライスで要素を指定して削除

del list_hoge[0]
del list_hoge[-1]
del list_hoge[0:5]
del list_hoge[:]

内包表記で削除(フィルタリング)する方法もある

list_fuga = [item for item in list_hoge if re.search('^[a-z]+$', item)]

文字列化

str_list = '/'.join(list_hoge)  # 0/1/2/3/4

ファイル操作

<br>

ディレクトリが無ければ作成

dir_path = 'path/to/directory'

if not os.path.exists(dir_path):
    os.makedirs(dir_path, exist_ok=True)

ファイル読み込み

全て読み込み

with open(filepath) as f:
    print(f.read())

最初の一行を読み込み

with open(filepath) as f:
    print(f.readline())

一行ずつ配列で読み込み

with open(filepath) as f:
    print(f.readlines())

ファイル書き込み

新規作成

with open(filepath, mode='x') as f:
    f.write(data)

上書き、存在しない場合は新規作成

with open(filepath, mode='w') as f:
    f.write(data)

追記、存在しない場合は新規作成

with open(filepath, mode='a') as f:
    f.write(data)

日時 datetime

<br>

インポート

import datetime
from pytz import timezone

現在時刻を、yyyymmdd形式で取得

date_now = datetime.datetime.now()
date_now = date_now.astimezone(timezone('Asia/Tokyo'))
ymd = date_now.strftime('%Y%m%d')

日時を作成

dt = datetime.datetime(2020, 10, 12, 17, 20, 00, 5000)  # ミリ秒まで指定可能
dt = datetime.datetime.strptime('2020/10/12 17:20', '%Y/%m/%d %H:%M')

日時から文字列に変換

dt_str = dt.strftime('%Y-%m-%d %H:%M:%S')

Unix時間に変換

unixtime = dt.timestamp()

日時から日や時を抽出

year        = dt.year
month       = dt.month
day         = dt.day
hour        = dt.hour
minute      = dt.minute
second      = dt.second
microsecond = dt.microsecond

日時同士の計算、日時を操作

dt1 = datetime.datetime.strptime('2020/01/01 12:00', '%Y/%m/%d %H:%M')
dt2 = datetime.datetime.strptime('2020/01/02 12:00', '%Y/%m/%d %H:%M')
td  = dt2 - dt1  # durationはtimedelta型
extra = td + datetime.timedelta(minutes=1)

範囲内かどうかを判定 時間有でもできるはず

sdate = datetime.datetime.strptime('2020/01/01', '%Y/%m/%d')
edate = datetime.datetime.strptime('2020/12/31', '%Y/%m/%d')
tgt_date = datetime.datetime.strptime('2020/06/01', '%Y/%m/%d')
if sdate <= tgt_date <= edate:
    print(tgt_date)

timedelta型から日、秒、マイクロ秒を取得（他は計算する必要がある）

td.days
td.seconds
td.microseconds

Pandas

<br>

インポート

import pandas as pd

標準出力設定 データフレームを省略せず表示

pd.set_option('display.max_columns', 99999)
pd.set_option('display.max_rows', 99999)

データフレームを作成

df = pd.DataFrame(data=None)          # 空のデータフレーム
df = pd.DataFrame(data=other_series)  # Seriesから作成
df = pd.DataFrame(data=other_list)    # 配列から作成
df = pd.DataFrame(data=other_np_arr)  # numpy配列から作成
df = pd.read_csv('path/to/csvfile')   # CSVファイルから作成
df = pd.read_csv('path/to/csvfile', header=None)  # Headerがない場合

データフレームにデータフレームをappend

df = append(df_other)
df = pd.concat([df, df_other])
df = pd.marge(df, df_other, on=col)

appendやsortをしてバラバラになったindexを振りなおす

df.reset_index(drop=True)

行・列を抽出 (そのままmatplotlibとかに使える)

data = df.loc['value']     # 行
data = df.loc[:, 'value']  # 列
data = df.iloc[1]     # 行
data = df.iloc[:, 1]  # 列

行・列ごとの差分を取得

diff = df.diff()        # 1行ごとの差分を取得
diff = df.diff(2)       # 2行ごとの差分を取得
diff = df.diff(axis=1)  # 1列ごとの差分を取得

行・列を削除

df = df.drop('row_name', axis=0)  # 行
df = df.drop('col_name', axis=1)  # 列
df = df.drop(['col1,', 'col2', 'col3'], axis=1)  # リストしていも可

重複削除

df = df.drop_duplicates()

集約

df = df.groupby('Date').mean()

まとめてdatetime型に変換

pd.to_datetime(df['Datetime'])

date列のフォーマットを変換

df = df['Date'].dt.strftime('%Y%m%d')

年月日抽出（seriesで返る）

df = df['Date'].dt.year          # 年
df = df['Date'].dt.month         # 月
df = df['Date'].dt.day           # 日
df = df['Date'].dt.hour          # 時
df = df['Date'].dt.minute        # 分
df = df['Date'].dt.second        # 秒
df = df['Date'].dt.weekday_name  # 曜日(文字列)
df = df['Date'].dt.dayofweek     # 曜日(数値)

列をインデックスに割り当て

df = df.set_index('time')

CSV出力

df.to_csv(filepath)

a列とb列のみ出力、ヘッダとインデックス無し

df.to_csv(filepath, columns=['a', 'b'], header=False, index=False)

ax.set_xticklabels(['国語', '算数', '理科', '社会'])
